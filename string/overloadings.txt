        static dynamic IndexOf(string word)
        {

            char letter = 'z';
            int index = 0;

            for (int i = 0; i < word.Length; i++)
            {
                if (word[i] == letter)
                {
                    index = i;
                    Console.WriteLine(i);
                    break;
                }

            }
            return -1;
        }
        static dynamic IndexOf(string word, int num)
        {

            char letter = 'a';
            int index = 0;

            if (num > word.Length)
            {
                Console.WriteLine("adam kimi eded yaz");
            }
            else
            {
                for (int i = num; i < word.Length; i++)
                {
                    if (word[i] == letter)
                    {
                        index = i;
                        Console.WriteLine(i);
                        break;
                    }

                }
            }
            return 0;
        }

        static dynamic IndexOf(string word, int num, int count)
        {

            char letter = 'z';
            int index = 0;
            if (count <= num || count > word.Length)
            {
                Console.WriteLine("adam kimi eded yaz");
            }
            else
            {
                for (int i = num; i < count; i++)
                {
                    if (word[i] == letter)
                    {
                        index = i;
                        Console.WriteLine(i);
                        break;
                    }


                }
            }
            return 0;
        }
